SELECT 
       PCPEDI.NUMPED, 
	   PCPEDI.CODPROD, 
	   PCPRODUT.DESCRICAO, 
	   PCPRODUT.UNIDADE, 
	   PCPEDI.QT, 
	   PCPEDI.PVENDA AS PRECO_UNIT, 
	   PCPEDI.ALIQICMS1,
	   PCPEDI.ALIQICMS2, 
	   PCPEDI.PERCIPI, 
	   PCPEDI.VLIPI 
FROM   PCPEDI
	   INNER JOIN PCPEDC ON PCPEDI.NUMPED = PCPEDC.NUMPED
	   INNER JOIN PCPRODUT ON PCPRODUT.CODPROD = PCPEDI.CODPROD
WHERE PCPEDC.NUMPED = 5000587 
  AND (PCPEDC.POSICAO = 'F')
